- docker logs mockup-app1 | rg -i "redis|connection|timeout|refused|error"
- docker logs mockup-app2 | rg -i "redis|connection|timeout|refused|error"
- docker logs mockup-app3 | rg -i "redis|connection|timeout|refused|error"
